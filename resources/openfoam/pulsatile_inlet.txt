type            codedFixedValue;
value           uniform ({u_mean_x} {u_mean_y} 0);
name            pulsatileInlet;
code
#{{{{
    const scalar t = this->db().time().value();
    const scalar amp = {amp};
    const scalar freq = {freq};
    const scalar shift = {shift};
    const scalar pi = constant::mathematical::pi;

    // Assuming variation scales the magnitude
    scalar scale = 1.0 + amp * sin(2*pi*freq*t + shift);

    operator== (vector({u_mean_x}, {u_mean_y}, 0) * scale);
